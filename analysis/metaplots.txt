#To separate the different contexts use:
zcat trim.MANC1_1_bismark_bt2_pe.CX_report.txt.gz | \
awk 'BEGIN{OFS="\t"}
{
  total = $4 + $5
  if (total > 0) {
    start = $2 - 1
    end   = $2
    perc  = 100 * $4 / total

    if ($6 == "CG")
      print $1, start, end, perc >> "CG.bedGraph"
    else if ($6 == "CHG")
      print $1, start, end, perc >> "CHG.bedGraph"
    else if ($6 == "CHH")
      print $1, start, end, perc >> "CHH.bedGraph"
  }
}'

#sort the individual constexts

LC_COLLATE=C sort -k1,1 -k2,2n CG.bedGraph > CG.sorted.bedGraph
LC_COLLATE=C sort -k1,1 -k2,2n CHG.bedGraph > CHG.sorted.bedGraph
LC_COLLATE=C sort -k1,1 -k2,2n CHH.bedGraph > CHH.sorted.bedGraph

bedGraphToBigWig CG.sorted.bedGraph /dir/Neurosporasizes.genome CG_bismark.bw
bedGraphToBigWig CHG.sorted.bedGraph /dir/Neurosporasizes.genome CHG_bismark.bw
bedGraphToBigWig CHH.sorted.bedGraph /dir/Neurosporasizes.genome CHH_bismark.bw

computeMatrix reference-point --referencePoint TSS -S CG_bismark.bw CHG_bismark.bw CHH_bismark.bw -R /dir/Neurospora_crassa.NC12_gff3.bed -a 2000 -b 2000 -p 4 -o matrix_contexts_tss.mat.gz
